CREATE DATABASE uiu_healthcare;

USE DATABASE uiu_healthcare;

CREATE TABLE students (
    StudentID VARCHAR(20) PRIMARY KEY,
    Email VARCHAR(100) UNIQUE NOT NULL,
    FullName VARCHAR(100) NOT NULL,
    Age INT CHECK (Age BETWEEN 18 AND 40),
    Gender ENUM('Male', 'Female', 'Other') NOT NULL,
    PasswordHash VARCHAR(255) NOT NULL
);

CREATE TABLE medical_diagnose_data (
    id INT AUTO_INCREMENT PRIMARY KEY,
    disease VARCHAR(255),
    symptoms TEXT,
    first_aid_tips TEXT,
    tutorial_link VARCHAR(500),
    age INT,
    gender ENUM('Male', 'Female', 'Other'),
    weight_kg FLOAT,
    recommended_specialist VARCHAR(255)
);

LOAD DATA INFILE 'C:/xampp/htdocs/UIU-Health-Care/medical_diagnose_data.csv'
INTO TABLE medical_diagnoses
FIELDS TERMINATED BY ',' 
ENCLOSED BY '"'
